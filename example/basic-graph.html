📝 当前代码

// 初始化
const graph = new Graph({
  container: '#app'
});

// 注册插件
graph.use(new DragNodePlugin());
graph.use(new SelectNodePlugin());

// 添加节点
graph.addNode({
  type: 'start',
  x: 100,
  y: 100,
  label: '开始'
});

// 监听事件
graph.on('node:click', ({ id }) => {
  console.log('节点:', id);
});
